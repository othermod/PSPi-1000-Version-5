# uncomment to force a console size. By default it will be display's size minus
# overscan.
framebuffer_width=800
framebuffer_height=480

#uncomment to overclock the arm. 700 MHz is the default.
#arm_freq=800

#Disable eeprom read at bootup
force_eeprom_read=0

#Set up I2C-1 on CSI pins 44 and 45
#was using I2C-0, but muxing was causing issues and the i2c0 overlay isn't functioning
dtoverlay=i2c1,pins_44_45

# Enable audio (loads snd_bcm2835)
dtparam=audio=on

# Enable DRM VC4 V3D driver on top of the dispmanx display stack
dtoverlay=vc4-fkms-v3d
max_framebuffers=2

#enable USB
dtoverlay=dwc2,dr_mode=host

#does this do anything?
#DISABLE_HDMI=1

#temporary while using v4 board to test
gpio=27=op,dh

#configure DPI
dtoverlay=lcd
overscan_left=0
overscan_right=0
overscan_top=0
overscan_bottom=0
enable_dpi_lcd=1
display_default_lcd=1
dpi_group=2
dpi_mode=87
dpi_output_format=458774
dpi_timings=800 0 40 48 88 480 0 13 3 32 0 0 0 60 0 32000000 6

#PWM mono audio from GPIO 18
dtoverlay=audremap,pins_18_19

#Does not work on CM4
#dtoverlay=pwm,pin=18,func=2

#Reduce PWM Audio Noise
#verify it still works with CM4 and Buster
disable_audio_dither=1
